### Files to create

**1. `app/services/chat_service.rb`**

```ruby
class ChatService
  class Error < StandardError; end

  DEFAULT_MODEL = "anthropic/claude-sonnet-4.6"
  SYSTEM_PROMPT = "You are a helpful assistant. Be concise."

  def initialize(history: [], model: DEFAULT_MODEL)
    @history = history
    @model = model
  end

  def reply(user_message)
    chat = RubyLLM.chat(model: @model, provider: :openrouter)
    chat.with_instructions(SYSTEM_PROMPT)
    @history.each do |msg|
      chat.add_message(role: msg[:role].to_sym, content: msg[:content])
    end
    response = chat.ask(user_message)
    response.content
  rescue RubyLLM::Error => e
    raise Error, "LLM call failed: #{e.message}"
  end
end
```

CRITICAL: use ONLY these real RubyLLM 1.14 methods:
- `RubyLLM.chat(model:, provider:)` returns Chat
- `chat.with_instructions(text)` sets system prompt
- `chat.add_message(role:, content:)` appends a turn — Ruby parses kwargs as positional hash, this is correct
- `chat.ask(message)` sends turn, returns response
- `response.content` extracts assistant text

DO NOT use `RubyLLM::Client.new`, `chat.user(...)`, `chat.assistant(...)`, `chat.send_message(...)`, `RubyLLM.chat(messages: [...])` batch form, or `response.text`. Those don't exist in the gem.

**2. `app/models/message.rb`** (PORO Struct, NOT ActiveRecord — there is no DB)

```ruby
class Message
  ROLES = %w[user assistant].freeze
  attr_reader :role, :content

  def initialize(role:, content:)
    raise ArgumentError, "invalid role" unless ROLES.include?(role.to_s)
    @role = role.to_s
    @content = content.to_s
  end

  def to_h = { role: role, content: content }
  def user? = role == "user"
  def assistant? = role == "assistant"
end
```

**5. `app/controllers/messages_controller.rb`**

```ruby
class MessagesController < ApplicationController
  MAX_MESSAGES = 25

  def create
    user_text = params.fetch(:content, "").to_s.strip
    if user_text.empty?
      head :no_content and return
    end
    history = (session[:chat_history] || []).map { |m| { role: (m["role"] || m[:role]).to_sym, content: m["content"] || m[:content] } }
    user_message = Message.new(role: "user", content: user_text)
    begin
      assistant_text = ChatService.new(history: history).reply(user_text)
      assistant_message = Message.new(role: "assistant", content: assistant_text)
      append_to_session(user_message, assistant_message)
      respond_to do |format|
        format.turbo_stream do
          render turbo_stream: [
            turbo_stream.append("messages", partial: "messages/message", locals: { message: user_message }),
            turbo_stream.append("messages", partial: "messages/message", locals: { message: assistant_message }),
            turbo_stream.replace("composer", partial: "messages/form"),
          ]
        end
      end
    rescue ChatService::Error => e
      Rails.logger.warn("ChatService error: #{e.message}")
      render turbo_stream: turbo_stream.replace("composer", partial: "messages/form", locals: { error: e.message }), status: :unprocessable_entity
    end
  end

  private

  def append_to_session(*new_messages)
    history = session[:chat_history] || []
    new_messages.each { |m| history << m.to_h.transform_keys(&:to_s) }
    session[:chat_history] = history.last(MAX_MESSAGES)
  end
end
```

**11. `app/javascript/controllers/composer_controller.js`**

```js
import { Controller } from "@hotwired/stimulus"

export default class extends Controller {
  static targets = ["input", "submit"]

  submitOnEnter(event) {
    if (event.key === "Enter" && !event.shiftKey) {
      event.preventDefault()
      this.element.requestSubmit()
    }
  }

  autoresize() {
    this.inputTarget.style.height = "auto"
    this.inputTarget.style.height = `${this.inputTarget.scrollHeight}px`
  }

  disable() {
    if (this.hasSubmitTarget) this.submitTarget.disabled = true
  }
}
```

**12. `app/javascript/controllers/scroll_controller.js`**

```js
import { Controller } from "@hotwired/stimulus"

export default class extends Controller {
  connect() {
    this.observer = new MutationObserver(() => this.scrollToBottom())
    this.observer.observe(this.element, { childList: true, subtree: true })
    this.scrollToBottom()
  }

  disconnect() { this.observer?.disconnect() }

  scrollToBottom() {
    requestAnimationFrame(() => {
      this.element.scrollTop = this.element.scrollHeight
    })
  }
}
```
