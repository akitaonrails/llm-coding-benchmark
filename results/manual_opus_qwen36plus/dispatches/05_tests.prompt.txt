**2. `test/services/chat_service_test.rb`** — exercises the real `chat.ask` round-trip via WebMock

```ruby
require "test_helper"

class ChatServiceTest < ActiveSupport::TestCase
  setup do
    ENV["OPENROUTER_API_KEY"] = "test-key"
  end

  test "reply returns assistant content from OpenRouter" do
    stub_openrouter_chat(assistant_text: "Hello there!")
    service = ChatService.new
    assert_equal "Hello there!", service.reply("hi")
    assert_requested :post, "https://openrouter.ai/api/v1/chat/completions"
  end

  test "reply replays history into the request body" do
    stub_openrouter_chat(assistant_text: "Sure.")
    history = [
      { role: :user, content: "First message" },
      { role: :assistant, content: "First reply" }
    ]
    service = ChatService.new(history: history)
    service.reply("Second message")
    assert_requested :post, "https://openrouter.ai/api/v1/chat/completions" do |req|
      payload = JSON.parse(req.body)
      messages = payload["messages"]
      contents = messages.map { |m| m["content"] }
      contents.include?("First message") &&
        contents.include?("First reply") &&
        contents.include?("Second message")
    end
  end

  test "raises ChatService::Error on RubyLLM::Error" do
    stub_request(:post, "https://openrouter.ai/api/v1/chat/completions")
      .to_return(status: 503, body: "{\"error\":{\"message\":\"upstream busy\"}}")
    service = ChatService.new
    assert_raises(ChatService::Error) { service.reply("hi") }
  end
end
```

class MessagesControllerTest < ActionDispatch::IntegrationTest
  setup do
    ENV["OPENROUTER_API_KEY"] = "test-key"
  end

  test "POST /messages with text returns turbo stream containing both user and assistant bubbles" do
    stub_openrouter_chat(assistant_text: "Pong.")
    post messages_url, params: { content: "Ping?" }, as: :turbo_stream
    assert_response :success
    assert_match "Ping?", @response.body
    assert_match "Pong.", @response.body
  end

  test "POST /messages with blank content returns no_content" do
    post messages_url, params: { content: "  " }, as: :turbo_stream
    assert_response :no_content
  end

  test "POST /messages handles ChatService::Error gracefully" do
    stub_request(:post, "https://openrouter.ai/api/v1/chat/completions").to_return(status: 500, body: "{}")
    post messages_url, params: { content: "Hi?" }, as: :turbo_stream
